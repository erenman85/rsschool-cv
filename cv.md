const bubbleSort = (items)  => {<br>
  let stepsCount = items.length-1;<br>
  let swapped;<br>
  do {<br>
    swapped = false;<br>
    for (let i = 0; i < stepsCount; i += 1) {<br>
      if (items[i] > items[i + 1]) {<br>
        let temp = items[i];<br>
        items[i] = items[i + 1];<br>
        items[i + 1] = temp;<br>
      }<br>
    }<br>
    stepsCount -= 1;<br>
  } while (swapped); <br>
  return items;<br>
};<br>
